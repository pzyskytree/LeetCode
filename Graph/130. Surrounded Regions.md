## 130. Surrounded Regions

Medium

Given a 2D board containing `'X'` and `'O'` (**the letter O**), capture all regions surrounded by `'X'`.

A region is captured by flipping all `'O'`s into `'X'`s in that surrounded region.

**Example:**

```
X X X X
X O O X
X X O X
X O X X
```

After running your function, the board should be:

```
X X X X
X X X X
X X X X
X O X X
```

**Explanation:**

Surrounded regions shouldnâ€™t be on the border, which means that any `'O'` on the border of the board are not flipped to `'X'`. Any `'O'` that is not on the border and it is not connected to an `'O'` on the border will be flipped to `'X'`. Two cells are connected if they are adjacent cells connected horizontally or vertically.

//DFS
//Time: O(M * N) Space:O(1)
class Solution {
    int[][] dirs = new int[][]{{1, 0}, {-1, 0}, {0, 1}, {0, -1}};
    public void solve(char[][] board) {
        if (board == null || board.length == 0)
            return;
        int m = board.length, n = board[0].length;
        for (int i = 0; i < m; i++){
            if (board[i][0] == 'O'){
                dfs(board, i, 0);
            }
            if (board[i][n - 1] == 'O'){
                dfs(board, i, n - 1);
            }
        }
        for (int j = 0; j < n; j++){
            if (board[0][j] == 'O'){
                dfs(board, 0, j);
            }
            if (board[m - 1][j] == 'O'){
                dfs(board, m - 1, j);
            }
        }
        for (int i = 0; i < m; i++){
            for (int j = 0; j < n; j++){
                if (board[i][j] == 'O'){
                    board[i][j] = 'X';
                }
                else if (board[i][j] == '*'){
                    board[i][j] = 'O';
                }
            }
        }
    }
    public void dfs(char[][] board, int i, int j){
        if (i < 0 || i >= board.length || j < 0 || j >= board[0].length || board[i][j] != 'O')
            return;
        board[i][j] = '*';
        for (int k = 0; k < dirs.length; k++){
            dfs(board, i + dirs[k][0], j + dirs[k][1]);
        }
    }
}

                bfs(board, 0, j);
            }
            if (board[m - 1][j] == 'O'){
                bfs(board, m - 1, j);
            }
        }
        for (int i = 0; i < m; i++){
            for (int j = 0; j < n; j++){
                if (board[i][j] == 'O'){
                    board[i][j] = 'X';
                }
                else if (board[i][j] == '*'){
                    board[i][j] = 'O';
                }
            }
        }
    }
    public void bfs(char[][] board, int i, int j){
        int m = board.length, n = board[0].length;
        Queue<Integer> queue = new LinkedList<>();
        queue.offer(i * n + j);
        board[i][j] = '*';
        while (!queue.isEmpty()){
            int pos = queue.poll();
            for (int k = 0; k < dirs.length; k++){
                int newI = pos / n + dirs[k][0], newJ = pos % n + dirs[k][1];
                if (newI >= 0 && newI < m && newJ >= 0 && newJ < n && board[newI][newJ] == 'O'){
                    queue.offer(newI * n + newJ);
                    board[newI][newJ] = '*';
                }
            }
        }
    }
}

                self.bfs(board, m - 1, j)
        for i in range(m):
            for j in range(n):

                    board[i][j] = 'O'
                elif board[i][j] == 'O':
                    board[i][j] = 'X'
                    
    def bfs(self, board, i, j):
        queue = deque()
        m, n = len(board), len(board[0])
        queue.append(i * n + j)
        board[i][j] = '*'
        while queue:
            pos = queue.popleft()
            for dir in self.dirs:
                newI, newJ = int(pos / n) + dir[0], int(pos % n) + dir[1]
                if newI >= 0 and newI < m and newJ >= 0 and newJ < n and board[newI][newJ] == 'O':
                    queue.append(newI * n + newJ)
                    board[newI][newJ] = '*'

//Union Find:
//Time:O(M * N) Space: O(M * N)
class Solution {
    class UnionFind{
        int[] rank;
        int[] parent;
        public UnionFind(int n){
            rank = new int[n];
            parent = new int[n];
            for (int i = 0; i < n; i++){
                parent[i] = i;
            }
        }
        public int find(int p){
            if (parent[p] != p){
                parent[p] = find(parent[p]);
            }
            return parent[p];
        }
        public void union(int i, int j){
            int p1 = find(i);
            int p2 = find(j);
            if (p1 != p2){
                if (rank[p1] >= rank[p2]){
                    parent[p2] = p1;
                    if (rank[p1] == rank[p2]){
                        rank[p1]++;
                    }
                }else{
                    parent[p1] = p2;
                }
            }
        }
    }
    int[][] dirs = new int[][]{{0, 1}, {0, -1}, {1, 0}, {-1, 0}};
    public void solve(char[][] board) {
        if (board == null || board.length == 0)
            return;
        int m = board.length, n = board[0].length;
        UnionFind uf = new UnionFind(n * m + 1);
        for (int i = 0; i < m; i++){
            if (board[i][0] == 'O'){
                uf.union(0, i * n + 1);
            }
            if (board[i][n - 1] == 'O'){
                uf.union(0, i * n + n);
            }
        }
        for (int j = 0; j < n; j++){
            if (board[0][j] == 'O'){
                uf.union(0, j + 1);
            }
            if (board[m - 1][j] == 'O'){
                uf.union(0, (m - 1) * n + j + 1);
            }
        }
        for (int i = 0; i < m; i++){
            for (int j = 0; j < n; j++){
                if (board[i][j] == 'O'){
                    for (int k = 0; k < dirs.length; k++){
                        int newI = i + dirs[k][0], newJ = j + dirs[k][1];
                        if (newI >= 0 && newI < m && newJ >= 0 && newJ < n
                            && board[newI][newJ] == 'O'){
                            uf.union(i * n + j + 1, newI * n + newJ + 1);
                        }
                    }
                }
            }
        }
        for (int i = 0; i < m; i++){
            for (int j = 0; j < n; j++){
                if (uf.find(0) != uf.find(i * n + j + 1)){
                    board[i][j] = 'X';
                }
            }
        }
    }
}
